        TargetCompID,  
        ConfigItemsId,  
        CreateTime,  
        ModifieTime,  
        LastOperator,  
        OMSIdentifier,  
        BypassMargin,

        @CTCLACCOUNTNAME,  
        @SENDERCOMPID,  
        @TARGETCOMPID,  
        @CONFIGITEMSID,  
        @CREATETIME,  
        @MODIFIETIME,

        @BypassMargin,  
        @IsDropCopy,

END
 
 
ALTER PROCEDURE [dbo].[spUpdateCTCL]
    @CTCLId         INT,
    @CTCLAccountId  VARCHAR(45),
    @CTCLAccountName VARCHAR(45),
    @SenderCompID   VARCHAR(45),
    @TargetCompID   VARCHAR(45),
    @configItemsID  INT,
    @CreateTime     DATETIME,
    @ModifieTime    DATETIME,
    @LastOperator   VARCHAR(100),
    @OMSIdentifier  VARCHAR(MAX),
    @BypassMargin   BIT,
    @IsDropCopy     BIT,
    @Hedging        BIT,       
    @ClearingId     VARCHAR(100)
AS
BEGIN
    UPDATE CTCL
    SET
        CTCLAccountId   = @CTCLAccountId,
        CTCLAccountName = @CTCLAccountName,
        SenderCompID    = @SenderCompID,
        TargetCompID    = @TargetCompID,
        configItemsID   = @configItemsID,
        CreateTime      = @CreateTime,
        ModifieTime     = @ModifieTime,
        LastOperator    = @LastOperator,
        OMSIdentifier   = @OMSIdentifier,
        BypassMargin    = @BypassMargin,
        IsDropCopy      = @IsDropCopy,
        Hedging         = @Hedging,       
        ClearingId      = @ClearingId
    WHERE CTCLId = @CTCLId
END
